Here we look at lambda expressions. We can use boost lambda to initialize elements in a list or vector as follows

#include<boost/lambda/lambda.hpp>
#include<iostream>
int main(){
	std::list<int> v(10);
	for_each(
		v.begin(),
		v.end(),
		boost::lambda::_1=1
	);
	for(
		std::list<int>::iterator it=v.begin();
		it!=v.end();
		++it
	){
		std::cout<<(*it)<<std::endl;
	}

	return 0;
}

boost::lambda::_1 creates a lambda functor which assigns the value 1 to every element in v.
